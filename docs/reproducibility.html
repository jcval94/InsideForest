Reproducibility

  • Set random_state for deterministic results.
  • Record package versions including insideforest.
  • Seed Python and NumPy RNGs.
import random, numpy as np
from InsideForest import InsideForestClassifier
random.seed(0); np.random.seed(0)
clf = InsideForestClassifier(random_state=0)
